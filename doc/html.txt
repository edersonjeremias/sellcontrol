<script>
let allClients=[];let currentOriginalName=null;let currentFocus=-1;let clienteSelecionadoDetalhes=null;
function enviarWhatsApp(){let whats=document.getElementById('whatsapp').value.replace(/\D/g,'');let nome=document.getElementById('nome').value.trim();let senha=document.getElementById('senha').value.trim();if(!whats){showToast('Preencha o WhatsApp!',true);return;}if(!nome){showToast('Preencha o Nome!',true);return;}let phoneLink=whats;if((whats.length===10||whats.length===11)&&!whats.startsWith('55')){phoneLink='55'+whats;}const LINK_PAINEL="https://script.google.com/macros/s/AKfycbyfd9DwRoLPop81jfZEyi-VNenjwVcIAwy6HZaH02-VzCjxWM_KaGxGe1I-hD8sdis_/exec";let txtSenha=senha?senha:"Não cadastrada";let msg="Olá, *"+nome+"*! Tudo bem?\n\n"+"Segue o link exclusivo para acessar a sua *Sacolinha Virtual* e o seu Painel de Cliente da VM Kids:\n\n"+"Acesso: "+LINK_PAINEL+"\n"+"Senha Temporária: "+txtSenha+"\n\n"+"Lá você pode completar o seu cadastro, visualizar as peças que já estão garantidas na sua sacolinha, acompanhar as entregas, verificar pagamentos em aberto e, claro, encerrar a sua sacolinha quando quiser receber suas peças!\n\n"+"_Lembrando que você pode alterar sua senha a qualquer momento diretamente no seu painel._\n\n"+"Qualquer dúvida, estamos à disposição!";window.open("https://api.whatsapp.com/send?phone="+phoneLink+"&text="+encodeURIComponent(msg),'_blank');}
function showToast(msg,isError=false){let t=document.getElementById("custom-toast");document.getElementById("toast-msg").innerText=msg;let i=document.getElementById("toast-icon");i.innerText=isError?"error":"check_circle";i.style.color=isError?"var(--red)":"var(--green)";t.style.borderLeft=isError?"4px solid var(--red)":"4px solid var(--green)";t.className="show";setTimeout(()=>{t.className=t.className.replace("show","");},3000);}
function showLoader(s){document.getElementById('loader').style.display=s?'block':'none';}
function mascaraCpf(c){let v=c.value.replace(/\D/g,"");if(v.length>11)v=v.slice(0,11);v=v.replace(/(\d{3})(\d)/,"$1.$2");v=v.replace(/(\d{3})(\d)/,"$1.$2");v=v.replace(/(\d{3})(\d{1,2})$/,"$1-$2");c.value=v;}
function buscarCep(){let cep=document.getElementById('editCep').value.replace(/\D/g,'');if(cep.length===8){showLoader(true);fetch('https://viacep.com.br/ws/'+cep+'/json/').then(r=>r.json()).then(d=>{showLoader(false);if(!d.erro){document.getElementById('editRua').value=d.logradouro;document.getElementById('editBairro').value=d.bairro;document.getElementById('editCidade').value=d.localidade;document.getElementById('editEstado').value=d.uf;document.getElementById('editNum').focus();showToast("CEP preenchido.");}else{showToast("CEP não encontrado.",true);}}).catch(e=>{showLoader(false);showToast("Erro no CEP",true);});}}
const searchInput=document.getElementById('customClientInput');
searchInput.addEventListener('input',function(){closeAllLists();let v=this.value;if(!v){renderDropdown(allClients);return;}renderDropdown(allClients.filter(c=>c.nome.toLowerCase().includes(v.toLowerCase())));});
searchInput.addEventListener('click',function(){if(!document.getElementById('dropdown-list')){let v=this.value;renderDropdown(v?allClients.filter(c=>c.nome.toLowerCase().includes(v.toLowerCase())):allClients);}});
searchInput.addEventListener('keydown',function(e){let l=document.getElementById("dropdown-list");let i=l?l.getElementsByTagName("li"):[];if(e.key==="ArrowDown"){currentFocus++;addActive(i);}else if(e.key==="ArrowUp"){currentFocus--;addActive(i);}else if(e.key==="Enter"||e.key==="Tab"){if(currentFocus>-1&&i.length>0){e.preventDefault();i[currentFocus].click();document.getElementById('nome').focus();}}else if(e.key==="Escape"){closeAllLists();}});
function renderDropdown(d){closeAllLists();currentFocus=-1;if(d.length===0)return;let l=document.createElement("ul");l.setAttribute("id","dropdown-list");l.setAttribute("class","dropdown-list");document.getElementById('dropdownContainer').appendChild(l);d.forEach(c=>{let li=document.createElement("li");li.className="dropdown-item";li.innerHTML=c.nome+(c.bloqueado?"<span class='dropdown-badge'>BLOQ</span>":"");li.addEventListener("click",function(e){searchInput.value=c.nome;closeAllLists();loadClientInfo(c.nome);});l.appendChild(li);});}
function addActive(i){if(!i)return false;removeActive(i);if(currentFocus>=i.length)currentFocus=0;if(currentFocus<0)currentFocus=(i.length-1);i[currentFocus].classList.add("dropdown-item-active");i[currentFocus].scrollIntoView({block:'nearest'});}
function removeActive(i){for(let x=0;x<i.length;x++)i[x].classList.remove("dropdown-item-active");}
function closeAllLists(e){let x=document.getElementsByClassName("dropdown-list");for(let i=0;i<x.length;i++){if(e!=x[i]&&e!=searchInput)x[i].parentNode.removeChild(x[i]);}}
document.addEventListener("click",function(e){closeAllLists(e.target);});
document.getElementById('senha').addEventListener('keydown',function(e){if(e.key==="Enter"&&!e.shiftKey){e.preventDefault();saveData();}});

function refreshData(){showLoader(true);google.script.run.withSuccessHandler(r=>{if(!r){showLoader(false);showToast("Erro de leitura.",true);return;}allClients=r.data;showLoader(false);if(currentOriginalName){let ex=allClients.find(c=>c.nome===currentOriginalName);if(ex){searchInput.value=currentOriginalName;loadClientInfo(currentOriginalName);}else{fullReset();}}}).withFailureHandler(e=>{showLoader(false);showToast("Erro no servidor",true);}).getClientsData();}
function loadClientInfo(n){let b=document.getElementById('headerBadge');let btnB=document.getElementById('btnBlock');let btnU=document.getElementById('btnUnlock');let btnD=document.getElementById('btnDel');let btnE=document.getElementById('btnEdit');let btnZ=document.getElementById('btnZap');let c=allClients.find(x=>x.nome===n);if(c){currentOriginalName=c.nome;clienteSelecionadoDetalhes=c.detalhes;document.getElementById('nome').value=c.nome;document.getElementById('whatsapp').value=c.whatsapp;document.getElementById('senha').value=c.senha||"";document.getElementById('observacao').value=c.motivo;btnD.classList.remove('hidden');btnE.classList.remove('hidden');b.classList.remove('hidden');btnZ.classList.remove('hidden');if(c.bloqueado){b.innerText="BLOQUEADO";b.className="status-badge st-blocked";btnB.classList.add('hidden');btnU.classList.remove('hidden');}else{b.innerText="ATIVO";b.className="status-badge st-active";btnB.classList.remove('hidden');btnU.classList.add('hidden');}}}
function fullReset(){searchInput.value="";document.getElementById('nome').value="";document.getElementById('whatsapp').value="";document.getElementById('senha').value="";document.getElementById('observacao').value="";document.getElementById('headerBadge').classList.add('hidden');document.getElementById('btnBlock').classList.add('hidden');document.getElementById('btnUnlock').classList.add('hidden');document.getElementById('btnDel').classList.add('hidden');document.getElementById('btnEdit').classList.add('hidden');currentOriginalName=null;clienteSelecionadoDetalhes=null;document.getElementById('nome').focus();}
function abrirModalEditar(){if(!currentOriginalName||!clienteSelecionadoDetalhes)return;let d=clienteSelecionadoDetalhes;document.getElementById('editInstaFake').value=currentOriginalName;document.getElementById('editNomeCompleto').value=d.nomeCompleto||"";document.getElementById('editCpf').value=d.cpf||"";document.getElementById('editNasc').value=d.nasc||"";document.getElementById('editCep').value=d.cep||"";document.getElementById('editRua').value=d.rua||"";document.getElementById('editNum').value=d.num||"";document.getElementById('editComp').value=d.comp||"";document.getElementById('editBairro').value=d.bairro||"";document.getElementById('editCidade').value=d.cidade||"";document.getElementById('editEstado').value=d.estado||"";document.getElementById('editEmail').value=d.email||"";document.getElementById('modalEditar').classList.remove('hidden');}
function fecharModalEditar(){document.getElementById('modalEditar').classList.add('hidden');}
function fecharAoClicarFora(e){if(e.target.id==='modalEditar')fecharModalEditar();}
function salvarDetalhes(){if(!currentOriginalName)return;let em=document.getElementById('editEmail').value;if(em&&!em.includes('@')){showToast("E-mail inválido.",true);return;}let det={nomeCompleto:document.getElementById('editNomeCompleto').value,cpf:document.getElementById('editCpf').value,nasc:document.getElementById('editNasc').value,cep:document.getElementById('editCep').value,rua:document.getElementById('editRua').value,num:document.getElementById('editNum').value,comp:document.getElementById('editComp').value,bairro:document.getElementById('editBairro').value,cidade:document.getElementById('editCidade').value,estado:document.getElementById('editEstado').value,email:em};fecharModalEditar();showLoader(true);google.script.run.withSuccessHandler(m=>{showToast(m);refreshData();}).withFailureHandler(e=>{showLoader(false);showToast(e.message,true);}).saveClientDetails(currentOriginalName,det);}
function saveData(){let n=document.getElementById('nome').value;let w=document.getElementById('whatsapp').value.replace(/\D/g,'');let s=document.getElementById('senha').value;let o=document.getElementById('observacao').value;if(!n.trim()){showToast('Nome obrigatório!',true);return;}showLoader(true);google.script.run.withSuccessHandler(m=>{showToast(m);if(!currentOriginalName)currentOriginalName=n.trim();refreshData();fullReset();}).withFailureHandler(e=>{showLoader(false);showToast('Erro: '+e.message,true);}).saveClientData(currentOriginalName,n,w,o,s);}
function toggleBlock(sb){if(!currentOriginalName)return;showLoader(true);google.script.run.withSuccessHandler(m=>{showToast(m);refreshData();}).toggleBlockStatus(currentOriginalName,sb);}
function confirmDelete(){if(!currentOriginalName)return;if(confirm("Excluir "+currentOriginalName+"?")){showLoader(true);google.script.run.withSuccessHandler(m=>{showToast(m);currentOriginalName=null;fullReset();refreshData();}).deleteClient(currentOriginalName);}}
window.onload=refreshData;
